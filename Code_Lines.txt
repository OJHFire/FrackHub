User Registration
------------------------------------------------
CREATE TABLE OPS$2042387.Member(
  Id INT not NULL,
  Name varchar(25) NOT NULL,
  Surname varchar(25) NOT NULL,
  Address varchar(100) NOT NULL,
  Contact_Number varchar(11) NOT NULL,
  Email varchar(50) NOT NULL,
  Password varchar(25) NOT NULL,
  PRIMARY KEY(Id)
);
